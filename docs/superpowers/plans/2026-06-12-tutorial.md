# Tutorial System Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a standalone tutorial run accessible from the main menu that teaches all 8 core mechanics via a scripted sandbox with mandatory-highlight UI and input interception.

**Architecture:** A `TutorialManager` autoload sequences 9 steps by binding to EventBus signals one-shot; a `TutorialOverlay` CanvasLayer (added as a child of `main.tscn`) handles dark-panel masking, gold-border highlighting, and input interception. Tutorial mode is activated via `GameState.is_tutorial = true` before loading `main.tscn` — no separate scene file is needed. `Main.gd._ready()` calls `TutorialManager.start()` when the flag is set.

**Tech Stack:** Godot 4 GDScript, GUT test framework, existing EventBus autoload, CanvasLayer for overlay.

```gdscript
class_name TutorialOverlay
extends CanvasLayer

const DARK_COLOR := Color(0.0, 0.0, 0.0, 0.72)

@onready var _top: ColorRect = $TopDark
@onready var _bottom: ColorRect = $BottomDark
@onready var _left: ColorRect = $LeftDark
@onready var _right: ColorRect = $RightDark
@onready var _border: Panel = $HighlightBorder
@onready var _text_box: PanelContainer = $TextBox
@onready var _instruction: Label = $TextBox/VBox/InstructionLabel
@onready var _step_label: Label = $TextBox/VBox/StepLabel
@onready var _confirm_btn: Button = $ConfirmButton

var _highlight_rect: Rect2 = Rect2()
var _block_input: bool = false
var _highlight_node_path: String = ""

signal confirm_pressed

func _ready() -> void:
	for panel in [_top, _bottom, _left, _right]:
		panel.color = DARK_COLOR
	_confirm_btn.pressed.connect(func(): confirm_pressed.emit())
	_confirm_btn.hide()
	# Style the highlight border: gold outline
	var style := StyleBoxFlat.new()
	style.border_color = Color(0.94, 0.75, 0.25, 1.0)
	style.set_border_width_all(2)
	style.shadow_color = Color(0.94, 0.75, 0.25, 0.35)
	style.shadow_size = 6
	style.bg_color = Color(0, 0, 0, 0)
	_border.add_theme_stylebox_override("panel", style)

func show_step(step: Dictionary, step_index: int, total: int) -> void:
	_instruction.text = step["text"]
	_step_label.text = "步骤 %d / %d" % [step_index + 1, total]
	_block_input = step.get("block_outside_input", false)
	_highlight_node_path = step.get("highlight_node", "")
	_confirm_btn.hide()

	if _highlight_node_path == "":
		_hide_dark_panels()
		_center_text_box()
		visible = true
		return

	# Poll each frame until the target node is visible
	_highlight_rect = Rect2()
	set_process(true)
	visible = true

func show_complete() -> void:
	_hide_dark_panels()
	_block_input = false
	_highlight_rect = Rect2()
	_center_text_box()
	_confirm_btn.show()
	_confirm_btn.position = Vector2(
		get_viewport().get_visible_rect().size.x / 2.0 - 80.0,
		get_viewport().get_visible_rect().size.y / 2.0 + 20.0
	)
	visible = true

func hide_overlay() -> void:
	visible = false
	set_process(false)
	_highlight_rect = Rect2()
	_block_input = false

func _process(_delta: float) -> void:
	if _highlight_node_path == "":
		set_process(false)
		return
	var node = get_tree().current_scene.find_child(
		_highlight_node_path.trim_prefix("%"), true, false)
	if node == null or not (node is Control):
		return
	var ctrl := node as Control
	if not ctrl.visible:
		return
	var rect: Rect2 = ctrl.get_global_rect().grow(6.0)
	if rect == _highlight_rect:
		return
	_highlight_rect = rect
	_update_dark_panels(rect)
	_position_text_box(rect)
	set_process(false)

func _update_dark_panels(hl: Rect2) -> void:
	var vp := get_viewport().get_visible_rect().size
	_top.visible = true
	_top.position = Vector2.ZERO
	_top.size = Vector2(vp.x, hl.position.y)
	_bottom.visible = true
	_bottom.position = Vector2(0.0, hl.end.y)
	_bottom.size = Vector2(vp.x, vp.y - hl.end.y)
	_left.visible = true
	_left.position = Vector2(0.0, hl.position.y)
	_left.size = Vector2(hl.position.x, hl.size.y)
	_right.visible = true
	_right.position = Vector2(hl.end.x, hl.position.y)
	_right.size = Vector2(vp.x - hl.end.x, hl.size.y)
	_border.visible = true
	_border.position = hl.position
	_border.size = hl.size

func _hide_dark_panels() -> void:
	for node in [_top, _bottom, _left, _right, _border]:
		node.visible = false

func _center_text_box() -> void:
	var vp := get_viewport().get_visible_rect().size
	_text_box.position = Vector2(
		vp.x / 2.0 - 160.0,
		vp.y / 2.0 - 40.0
	)

func _position_text_box(hl: Rect2) -> void:
	var vp := get_viewport().get_visible_rect().size
	var box_h := 80.0
	var y_below := hl.end.y + 12.0
	var y_above := hl.position.y - box_h - 12.0
	var y := y_below if y_below + box_h < vp.y else y_above
	var x := clampf(hl.position.x, 8.0, vp.x - 320.0 - 8.0)
	_text_box.position = Vector2(x, y)

func _input(event: InputEvent) -> void:
	if not visible:
		return
	# ESC exits tutorial with confirmation
	if event is InputEventKey and event.pressed and event.keycode == KEY_ESCAPE:
		_prompt_exit()
		get_viewport().set_input_as_handled()
		return
	if not _block_input:
		return
	if _highlight_rect == Rect2():
		return
	if event is InputEventMouseButton or event is InputEventMouseMotion:
		if not _highlight_rect.has_point(event.position):
			get_viewport().set_input_as_handled()

func _prompt_exit() -> void:
	GameState.is_tutorial = false
	get_tree().change_scene_to_file("res://scenes/ui/loading_screen.tscn")
```

- [ ] **Step 3: Implement TutorialManager**

Replace `src/autoloads/TutorialManager.gd` with:

```gdscript
extends Node

var is_active: bool = false
var current_step: int = 0

var _steps: Array = []
var _overlay: TutorialOverlay = null
var _completion_callable: Callable

func _ready() -> void:
	_steps = TutorialSteps.get_steps()

func get_step_count() -> int:
	return _steps.size()

## Called by Main._ready() when GameState.is_tutorial is true.
func start(overlay: TutorialOverlay) -> void:
	_overlay = overlay
	is_active = true
	current_step = 0
	_setup_scenario()
	_overlay.confirm_pressed.connect(_on_confirm_pressed)
	_enter_step(0)

func stop() -> void:
	if not is_active:
		return
	is_active = false
	if _overlay != null:
		_overlay.hide_overlay()
	GameState.is_tutorial = false

func _setup_scenario() -> void:
	GameState.gold = 150
	# Enemies: GameLoop.spawn_enemies() is suppressed when is_tutorial.
	# TutorialManager calls _spawn_tutorial_enemies() instead after Main is ready.
	# Altar: set required slots to 1 via signal after scene fully loads.
	call_deferred("_deferred_scenario_setup")

func _deferred_scenario_setup() -> void:
	_spawn_tutorial_enemies()
	_reduce_altar_requirement()

func _spawn_tutorial_enemies() -> void:
	# Emit a signal that Main.gd listens to in order to spawn fixed enemies.
	# Main._spawn_tutorial_enemies() places enemies at tiles 3, 6, and 9.
	EventBus.tutorial_spawn_enemies.emit()

func _reduce_altar_requirement() -> void:
	# Signal Main to reduce altar slot count to 1.
	EventBus.tutorial_setup_altar.emit()

func _enter_step(index: int) -> void:
	current_step = index
	if index >= _steps.size():
		_on_all_steps_done()
		return

	var step: Dictionary = _steps[index]

	if step["id"] == "complete":
		_overlay.show_complete()
		return

	_overlay.show_step(step, index, _steps.size())

	var sig_name: String = step["complete_signal"]
	if sig_name == "":
		return

	_completion_callable = func(_a=null, _b=null, _c=null): _advance()
	EventBus.connect(sig_name, _completion_callable, CONNECT_ONE_SHOT)

func _advance() -> void:
	_enter_step(current_step + 1)

func _on_all_steps_done() -> void:
	_overlay.show_complete()

func _on_confirm_pressed() -> void:
	stop()
	get_tree().change_scene_to_file("res://scenes/ui/loading_screen.tscn")
```

```markdown
# Tutorial System

Responsible for teaching new players AttributeLoop's 8 unique mechanics via a 9-step scripted run.

## Key Components

- **TutorialManager** (autoload) — sequences steps, binds EventBus signals one-shot, sets up sandbox scenario
- **TutorialOverlay** (CanvasLayer in main.tscn) — dark panel masking, gold border highlight, text box, input interception
- **TutorialSteps** (static class) — pure data: array of 9 step dictionaries

## Execution Flow

1. Main menu Tutorial button sets `GameState.is_tutorial = true`, loads `main.tscn`
2. `Main._ready()` detects flag → calls `TutorialManager.start(overlay)`
3. TutorialManager: sets gold=150, triggers fixed enemy spawn and altar resize
4. Each step: overlay shows highlight + text; EventBus signal fires one-shot on completion
5. Step 9 (complete): overlay shows confirm button → pressed → back to main menu

## Dependencies

- `EventBus` — all step completion signals
- `GameState.is_tutorial` — gates GameLoop spawn suppression and game-over suppression
- `Main.gd` — receives `tutorial_spawn_enemies` / `tutorial_setup_altar` signals
```
